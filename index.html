Data & Methodology

Source model behind this dashboard
-- Shapes the fact table into the Month, Stream, Type, Amount format this dashboard reads
SELECT
    d.period_label  AS month,
    a.account_name  AS stream,
    a.account_type  AS type,       -- 'revenue' or 'expense'
    SUM(f.amount) AS amount
FROM fact_transactions f
JOIN dim_date d     ON f.date_key    = d.date_key
JOIN dim_account a  ON f.account_key = a.account_key
GROUP BY d.period_label, a.account_name, a.account_type
ORDER BY d.date_key;
// Net margin = net profit divided by total revenue for the current filter context
Net Margin % =
DIVIDE( [Net Profit], [Total Revenue], BLANK() )
// Compares revenue to the same period one year earlier using a date table
Revenue YoY % =
VAR CurrentRevenue = [Total Revenue]
VAR PriorRevenue =
    CALCULATE( [Total Revenue], SAMEPERIODLASTYEAR( 'Dim Date'[Date] ) )
RETURN
    DIVIDE( CurrentRevenue - PriorRevenue, PriorRevenue, BLANK() )
